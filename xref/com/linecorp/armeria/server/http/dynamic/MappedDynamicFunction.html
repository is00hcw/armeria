1   package com.linecorp.armeria.server.http.dynamic;
2   
3   import static java.util.Objects.requireNonNull;
4   
5   import java.util.Map;
6   
7   import com.linecorp.armeria.common.http.AggregatedHttpMessage;
8   import com.linecorp.armeria.common.http.DefaultHttpResponse;
9   import com.linecorp.armeria.common.http.HttpHeaderNames;
10  import com.linecorp.armeria.common.http.HttpHeaders;
11  import com.linecorp.armeria.common.http.HttpRequest;
12  import com.linecorp.armeria.common.http.HttpResponse;
13  import com.linecorp.armeria.common.http.HttpStatus;
14  import com.linecorp.armeria.server.ServiceRequestContext;
15  import com.linecorp.armeria.server.http.AbstractHttpService;
16  
17  /**
18   * {@link DynamicHttpFunction} with args.
19   */
20  final class MappedDynamicFunction extends AbstractHttpService {
21  
22      /**
23       * {@link DynamicHttpFunction} instance that will be invoked with given args.
24       */
25      private final DynamicHttpFunction function;
26  
27      /**
28       * Arguments, represented in Map of variable name to its value.
29       */
30      private final Map<String, String> args;
31  
32      MappedDynamicFunction(DynamicHttpFunction function, Map<String, String> args) {
33          this.function = requireNonNull(function, "function");
34          this.args = requireNonNull(args, "args");
35      }
36  
37      @Override
38      public HttpResponse serve(ServiceRequestContext ctx, HttpRequest req) throws Exception {
39          final DefaultHttpResponse res = new DefaultHttpResponse();
40          ctx.blockingTaskExecutor().execute(() -> {
41              try {
42                  Object ret = function.serve(ctx, req, args);
43                  if (!(ret instanceof HttpResponse)) {
44                      throw new IllegalStateException("Illegal return type: " + ret.getClass().getSimpleName());
45                  }
46                  AggregatedHttpMessage msg = ((HttpResponse) ret).aggregate().get();
47                  res.write(msg.headers());
48                  res.write(msg.content());
49              } catch (Throwable e) {
50                  final long current = System.currentTimeMillis();
51                  HttpHeaders headers = HttpHeaders.of(HttpStatus.INTERNAL_SERVER_ERROR)
52                                                   .setTimeMillis(HttpHeaderNames.DATE, current);
53                  res.write(headers);
54              } finally {
55                  res.close();
56              }
57          });
58          return res;
59      }
60  }
